203  void resolveCwSegmentConnections(const Ptr& ptrToMe)
204  {
205  // 1. Create all connections between carriageway segments, since these
206  // are required for connections between lanes and lane segments
207  for (auto& cwsme : m_cwSegmentsMap)
208  {
209  // pointer to segment (the object itself has to be edited, not a copy)
210  CarriageWaySegmentTemplatePtr cws = cwsme.second;
211  if (cws)
212  {
213  cws->setParent(ptrToMe);
214 
215  // link next element if id not 0 and existing
216  if ((cws->getNextId() != 0))
217  {
218  const auto itf = m_cwSegmentsMap.find(cws->getNextId());

223  }
