161  {

165  BoundingRectangle mBox = lane->getBoundingBox();

171  }
172  }
173

175  if (validLanes.empty())

177  return out;
178  }
179

181  // search valid Lane Segments
182  std::vector<LaneSegmentPtr> validLaneSegments;
183  for (const auto& lane : validLanes)
184  {
185  for (const auto& lsme : lane->getLaneSegmentsMap())
186  {
187  const LaneSegmentPtr& laneSeg{lsme.second};
188  const BoundingRectangle mBox{laneSeg->getBoundingBox()};
189 
190  if (mBox.checkInside(point))
191  {
192  validLaneSegments.push_back(laneSeg);
193  }
194  }
195  }
196 
197  // return if no lane segment was found
198  if (validLaneSegments.empty())
199  {
200  return out;
201  }
202 
203  //======================================
204  // now find the most suitable segment
205  LanePosition bestInside{};
206  LanePosition bestOutside{};
207  float minOutsideDist{0.0F};
208  bool bi{false};
209  bool bo{false};
210 
211  for (const auto& curSeg : validLaneSegments)
212  {
213  bool inside{false};
214  LanePosition curLP{calculateLanePosition(curSeg, point, inside)};
215 
216  // conditions intersection of projection is inside lane AND
217  // lateral distance is smaller than half of lane width in projection point AND
218  // the distance is closer than a previous found candidate
219  if (inside
220  && (std::abs(curLP.lateralDistance) <= (curLP.laneSegment->getWidth(curLP.longitudinalDistance) / 2)
221  + 0.001F) // add 1mm to avoid rounding errors
222  && (!bi || std::abs(curLP.lateralDistance) < std::abs(bestInside.lateralDistance)))
223  {
224  bestInside = curLP;
225  bi = true;
226  }
227  else if (!inside)
228  {
229  float startDis{std::abs(curLP.longitudinalDistance)};
230  float endDis{std::abs(curLP.longitudinalDistance - curSeg->getLength())};
231 
232  float useDis{startDis < endDis ? startDis : endDis};
233 
234  if (!bo || useDis < minOutsideDist)
235  {
236  bestOutside = curLP;
237  minOutsideDist = useDis;
238  bo = true;
239  }
240  }
241  }
242 
243  if (bi)
244  {
245  return bestInside;
246  }
247  else if (bo)
248  {
249  return bestOutside;
250  }
251 
252  return out;
253  }

480  {

487  Vector2<float> perpDir{Vector2<float>(-lineDir.getY(), lineDir.getX())};
488  Line2d<float> perp{Line2d<float>(p + perpDir * 15, p - perpDir * 15)}; // 15m in both directions
489 
490  Vector2<float> intersection;
491  Line2d<float>::IntersectingType ist{line.isIntersecting(perp, &intersection)};
492 
493  insideIntersection = (ist == Line2d<float>::LineIntersecting);
494 
495  LanePosition out{};

497  out.transformationMatrix = TransformationMatrix2d<float>(
498  static_cast<float>(point.getCourseAngleInRad()
499  - segment->getStartPoint().getWgsPoint().getCourseAngleInRad()),
