830 inline int16_t DeviceStatus6303::getContent(const ContentId cId, const T*& cData)
831 {
832  const int32_t idx = findContent(cId);
833  if (idx == -1)
834  {
835  cData = nullptr;
836  return -1;
837  }
838 
839  const ContentDescr& c = m_contentEntries.at(static_cast<uint32_t>(idx));
840  if (c.getElementType() == ElementType::STRUCT)
841  {
842  return -2;
843  }
844 
845  if (DeviceStatus6303::getElementType<T>(*cData) != c.getElementType())
846  {
847  cData = nullptr;
848  return -3;
849  }
850 
851  cData = reinterpret_cast<const T*>(c.getContentBuffer());
852  return c.getNbOfBytes();
853 }
