371 inline float normalizeRadians(float angleInRad)
372 {
373  if (std::abs(angleInRad) > (3.0F * static_cast<float>(M_PI)))
374  {
375  // For numerical stability we must use this sin/cos/atan2
376  // implementation even though it might consume more cycles.
377  // Note that radians = -pi converts to atan2(0,-1) = +pi!
378  angleInRad = std::atan2(std::sin(angleInRad), std::cos(angleInRad));
379  // radians now in (-pi,+pi]
380  } // if
381  else
382  {
383  // fast version for "normal" out of range values
384  while (angleInRad > static_cast<float>(M_PI))
385  {
386  angleInRad -= twoPif;
387  } // while
388  while (angleInRad <= static_cast<float>(-M_PI))
389  {
390  angleInRad += twoPif;
391  } // while
392  } // else
393  return angleInRad;
394 }
