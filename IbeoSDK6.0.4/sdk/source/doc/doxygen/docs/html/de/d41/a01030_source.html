109 public:
110  bool checkInside(const PositionWgs84& point) const
111  {
112  return (point.getLatitudeInDeg() <= this->maxLatitude && point.getLatitudeInDeg() >= this->minLatitude
113  && point.getLongitudeInDeg() >= this->minLongitude && point.getLongitudeInDeg() <= this->maxLongitude);
114  }
115 
116  void expand(const BoundingRectangle& other)
117  {
118  minLatitude = (other.minLatitude >= minLatitude) ? minLatitude : other.minLatitude;
119  maxLatitude = (other.maxLatitude <= maxLatitude) ? maxLatitude : other.maxLatitude;
120  minLongitude = (other.minLongitude >= minLongitude) ? minLongitude : other.minLongitude;
121  maxLongitude = (other.maxLongitude <= maxLongitude) ? maxLongitude : other.maxLongitude;
122  }
123 
124 public:
125  double minLongitude;
126  double maxLongitude;
127  double minLatitude;
128  double maxLatitude;
129 }; // BoundingRectangle
